458  *
459  * This function computes a coordinate based on a geometric stretching ratio.
460  * If the ratio (r) is 1.0, a uniform distribution is used:
461  * x(i) = L * (i/N)
462  *
463  * If r != 1.0, a geometric stretching is applied:
464  * x(i) = L * [ (r^(i/N) - 1 ) / (r - 1) ]
465  *
466  * Here:
467  * - i : The current index along the dimension.
468  * - N : The total number of divisions along that dimension.
469  * - L : The length of the domain along that dimension.
470  * - r : The stretching ratio. r > 1.0 stretches the grid in a geometric fashion
471  * increasing spacing away from the start, whereas 0 < r < 1.0 would
472  * cluster points near the start.
473  *
474  * @param[in] i Current index (0 <= i <= N).
475  * @param[in] N Number of segments along the dimension.
476  * @param[in] L Total length of the domain.
477  * @param[in] r Stretching ratio.
478  *
479  * @return PetscReal The computed coordinate at index i.
480  *
481  * @note This function does not return a PetscErrorCode because it
482  * does not allocate memory or call PETSc routines that can fail.
483  * It is just a helper calculation function.
484  **/
485 static inline PetscReal ComputeStretchedCoord(PetscInt i, PetscInt N, PetscReal L, PetscReal r)
486 {
487  if (N <=1) return 0.0;
488  PetscReal fraction = (PetscReal)i / ((PetscReal)N - 1.0);
489  if (PetscAbsReal(r - 1.0) < 1.0e-9) { // Use a tolerance for float comparison
490  return L * fraction;
491  } else {
492  return L * (PetscPowReal(r, fraction) - 1.0) / (r - 1.0);
493  }
494 }

551 /**
552  * @brief Reads physical coordinates from a file and populates the DMDA for a specific block.
553  *
554  * This function handles the collective read of an interleaved (X Y Z per line)
555  * multi-block grid file. It assumes the file header (nblk, dimensions) has
556  * already been processed by ReadGridFile.
557  *
558  * The process is robust for parallel execution:
559  * 1. Rank 0 opens the grid file.
560  * 2. It intelligently skips past the header section and the coordinate data
561  * for all blocks *preceding* the current block being processed (`user->_this`).
562  * 3. It then reads the entire coordinate data for the *current* block into
563  * a single contiguous buffer `gc`.
564  * 4. This global buffer `gc` is broadcast to all other MPI ranks.
565  * 5. Each rank then loops through its local (owned + ghost) node indices,
566  * calculates the corresponding index in the global `gc` array, and copies
567  * the (x,y,z) coordinates into its local PETSc coordinate vector.
568  *
569  * @param user The UserCtx for a specific block. Its `_this` field must be set,
570  * and its IM, JM, KM fields must be correctly pre-populated.
571  * @return PetscErrorCode 0 on success, or a PETSc error code on failure.
572  */
573 static PetscErrorCode ReadAndSetCoordinates(UserCtx *user, FILE *fd)
574 {
575  PetscErrorCode ierr;
576  SimCtx *simCtx = user->simCtx;
577  PetscMPIInt rank = simCtx->rank;
578  PetscInt block_index = user->_this;
579  PetscInt IM = user->IM, JM = user->JM, KM = user->KM;
580  DMDALocalInfo info;
581  Cmpnts ***coor;
582  Vec lCoor;
583  PetscReal *gc = NULL; // Global coordinate buffer, allocated on all ranks
584 
585  PetscFunctionBeginUser;
586 
587  PROFILE_FUNCTION_BEGIN;
588 
589  LOG_ALLOW(LOCAL, LOG_DEBUG, "Rank %d: Reading interleaved coordinates from file for block %d...\n",
590  simCtx->rank, block_index);
591 
592  // 1. Allocate the buffer on ALL ranks to receive the broadcast data.
593  // PetscInt n_nodes = (IM + 1) * (JM + 1) * (KM + 1);
594  PetscInt n_nodes = (IM) * (JM) * (KM);
595  ierr = PetscMalloc1(3 * n_nodes, &gc); CHKERRQ(ierr);
596 
597  // 2. Only Rank 0 opens the file and reads the data.
598  if (rank == 0) {
599  if (!fd) SETERRQ(PETSC_COMM_SELF, PETSC_ERR_FILE_OPEN, "Recieved a NULL file handle.\n");
600 
601  // Read the coordinate data for the CURRENT block.
602  for (PetscInt k = 0; k < KM; k++) {
603  for (PetscInt j = 0; j < JM; j++) {
604  for (PetscInt i = 0; i < IM; i++) {
605  PetscInt base_index = 3 * ((k * (JM) + j) * (IM) + i);
606  if (fscanf(fd, "%le %le %le\n", &gc[base_index], &gc[base_index + 1], &gc[base_index + 2]) != 3) {
607  SETERRQ(PETSC_COMM_SELF, PETSC_ERR_FILE_READ, "Error reading coordinates for node (i,j,k)=(%d,%d,%d) in block %d", i, j, k, block_index);
608  }
609  }
610  }
611  }
612 
613  }
614 
615  // 3. Broadcast the coordinate block for the current block to all other processes.
616  ierr = MPI_Bcast(gc, 3 * n_nodes, MPIU_REAL, 0, PETSC_COMM_WORLD); CHKERRQ(ierr);
617 
618  // 4. Each rank populates its local portion of the coordinate vector.
619  ierr = DMDAGetLocalInfo(user->da, &info); CHKERRQ(ierr);
620  ierr = DMGetCoordinatesLocal(user->da, &lCoor); CHKERRQ(ierr);
621  ierr = DMDAVecGetArray(user->fda, lCoor, &coor); CHKERRQ(ierr);
622 
623  for (PetscInt k = info.zs; k < info.zs + info.zm; k++) {
624  for (PetscInt j = info.ys; j < info.ys + info.ym; j++) {
625  for (PetscInt i = info.xs; i < info.xs + info.xm; i++) {
626  if(k< KM && j < JM && i < IM){
627  //PetscInt base_idx = 3 * ((k * (JM + 1) + j) * (IM + 1) + i);
628  PetscInt base_idx = 3 * ((k * (JM) + j) * (IM) + i);
629  coor[k][j][i].x = gc[base_idx];
630  coor[k][j][i].y = gc[base_idx + 1];
631  coor[k][j][i].z = gc[base_idx + 2];
632  }
633  }
634  }
635  }
636 
637  // 5. Clean up and restore.
638  ierr = DMDAVecRestoreArray(user->fda, lCoor, &coor); CHKERRQ(ierr);
639  ierr = PetscFree(gc); CHKERRQ(ierr);
640 
641  PROFILE_FUNCTION_END;
642 
643  PetscFunctionReturn(0);
644 }

649  * @brief Populates coarse grid coordinates by restricting from a fine grid.
650  *
651  * This function is a direct adaptation of the coordinate restriction loop
652  * from the legacy `MG_Initial` function. It ensures that the physical location
653  * of a coarse grid node is identical to its corresponding parent node on the
654  * fine grid. The mapping from coarse to fine index (`i` -> `ih`) is determined

658  * @param fine_user The UserCtx for the fine grid (source).
659  * @return PetscErrorCode
660  */
661 static PetscErrorCode RestrictCoordinates(UserCtx *coarse_user, UserCtx *fine_user)
662 {
663  PetscErrorCode ierr;

665  Cmpnts ***c_coor;
666  const Cmpnts ***f_coor; // Use const for read-only access
667  DMDALocalInfo c_info;
668  PetscInt ih, jh, kh; // Fine-grid indices corresponding to coarse-grid i,j,k
669 
670  PetscFunctionBeginUser;
671 
672  PROFILE_FUNCTION_BEGIN;
673 
674  LOG_ALLOW_SYNC(LOCAL, LOG_DEBUG, "Rank %d: Restricting coords from level %d to level %d for block %d\n",
675  fine_user->simCtx->rank, fine_user->thislevel, coarse_user->thislevel, coarse_user->_this);
676 
677  ierr = DMDAGetLocalInfo(coarse_user->da, &c_info); CHKERRQ(ierr);
678 
679  ierr = DMGetCoordinatesLocal(coarse_user->da, &c_lCoor); CHKERRQ(ierr);

686  PetscInt xs = c_info.xs, xe = c_info.xs + c_info.xm;
687  PetscInt ys = c_info.ys, ye = c_info.ys + c_info.ym;
688  PetscInt zs = c_info.zs, ze = c_info.zs + c_info.zm;

703  // respecting the semi-coarsening flags of the FINE grid's UserCtx.
704  ih = coarse_user->isc ? i : 2 * i;
705  jh = coarse_user->jsc ? j : 2 * j;
706  kh = coarse_user->ksc ? k : 2 * k;

709

711  }
712  }
713  }
714 
715  ierr = DMDAVecRestoreArray(coarse_user->fda, c_lCoor, &c_coor); CHKERRQ(ierr);
716  ierr = DMDAVecRestoreArrayRead(fine_user->fda, f_lCoor, &f_coor); CHKERRQ(ierr);
717 
718  // After populating the local portion, scatter to update ghost regions.
719  Vec c_gCoor;
720  ierr = DMGetCoordinates(coarse_user->da, &c_gCoor); CHKERRQ(ierr);

722  ierr = DMLocalToGlobalEnd(coarse_user->fda, c_lCoor, INSERT_VALUES, c_gCoor); CHKERRQ(ierr);
723  ierr = DMGlobalToLocalBegin(coarse_user->fda, c_gCoor, INSERT_VALUES, c_lCoor); CHKERRQ(ierr);
724  ierr = DMGlobalToLocalEnd(coarse_user->fda, c_gCoor, INSERT_VALUES, c_lCoor); CHKERRQ(ierr);
725 
726  PROFILE_FUNCTION_END;
727 
728  PetscFunctionReturn(0);
729 }
