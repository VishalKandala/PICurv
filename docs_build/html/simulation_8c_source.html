268 /**
269  * @brief Executes the main time-marching loop for the coupled Euler-Lagrange simulation.
270  *
271  * This function orchestrates the advancement of the simulation from the configured
272  * StartStep to the final step. It does NOT perform the initial t=0 setup, as that
273  * is handled by InitializeEulerianState and PerformInitialSetup in main().
274  *
275  * For each timestep, it performs the following sequence:
276  * 1. **Pre-Solver Actions:** Updates time-dependent boundary conditions (e.g., fluxin)
277  * and resets particle states for the new step.
278  * 2. **Eulerian Solve:** Calls the refactored legacy Flow_Solver to advance the
279  * entire fluid field by one time step.
280  * 3. **Lagrangian Update:** Executes the full particle workflow: advection based
281  * on the previous step's velocity, followed by settling (location/migration)
282  * in the new grid, and finally interpolation of the new fluid velocity.
283  * 4. **Two-Way Coupling:** Scatters particle data back to the grid to act as source
284  * terms for the subsequent time step.
285  * 5. **History & I/O:** Updates the solver's history vectors and writes output files
286  * at the specified frequency.
287  *
288  * @param user Array of UserCtx structures for the finest grid level.
289  * @return PetscErrorCode 0 on success, non-zero on failure.
290  */
291 PetscErrorCode AdvanceSimulation(SimCtx *simCtx)
292 {
293  PetscErrorCode ierr;
294  // Get the master context from the first block. All blocks share it.
295  UserCtx *user = simCtx->usermg.mgctx[simCtx->usermg.mglevels-1].user;
296 
297  // Retrieve control parameters from SimCtx for clarity
298  const PetscInt StartStep = simCtx->StartStep;
299  const PetscInt StepsToRun = simCtx->StepsToRun;
300  const PetscInt OutputFreq = simCtx->OutputFreq;
301  const PetscReal dt = simCtx->dt;
302 
303  // Variables for particle removal statistics
304  PetscInt removed_local_ob, removed_global_ob;
305  PetscInt removed_local_lost, removed_global_lost;
306 
307  PetscFunctionBeginUser;
308  PROFILE_FUNCTION_BEGIN;
309  LOG_ALLOW(GLOBAL, LOG_INFO, "Starting main time-marching loop: %d steps from step %d (t=%.4f), dt=%.4f\n",
310  StepsToRun, StartStep, simCtx->StartTime, dt);
311 
312  // --- Main Time-Marching Loop ---
313  for (PetscInt step = StartStep; step < StartStep + StepsToRun; step++) {
314 
315  // =================================================================
316  // 1. PRE-STEP SETUP
317  // =================================================================
318 
319  // Update simulation time and step counters in the master context
320  simCtx->step = step + 1;
321  simCtx->ti += simCtx->dt; //simCtx->StartTime + step * simCtx->dt;
322 
323  LOG_ALLOW(GLOBAL, LOG_INFO, "--- Advancing Step %d (To t=%.4f) ---\n", simCtx->step, simCtx->ti);
324 
325 
326  // For particles, reset their status to prepare for the new advection/location cycle
327  if (simCtx->np > 0) {
328  LOG_ALLOW(GLOBAL, LOG_DEBUG, "Resetting all particle statuses to NEEDS_LOCATION.\n");
329  ierr = ResetAllParticleStatuses(user); CHKERRQ(ierr);
330  }
331 
332  // =================================================================
333  // 2. EULERIAN SOLVER STEP
334  // =================================================================
335 
336  LOG_ALLOW(GLOBAL, LOG_INFO, "Updating Eulerian Field ...\n");
337  if(strcmp(simCtx->eulerianSource,"load")==0){
338  //LOAD mode: Read pre-computed fields for the current step.
339  LOG_ALLOW(GLOBAL,LOG_INFO,"Eulerian Source 'load': Reading fields (t=%.4f,step=%d)...",simCtx->ti,simCtx->step);
340  for(PetscInt bi = 0; bi < simCtx->block_number;bi++){
341  ierr = ReadSimulationFields(&user[bi],simCtx->step); CHKERRQ(ierr);
342  }
343  }else if(strcmp(simCtx->eulerianSource,"solve")==0){
344  // SOLVE mode:Call the refactored, high-level legacy solver. This single function
345  // advances the entire multi-block fluid field from t_n to t_{n+1}.
346  LOG_ALLOW(GLOBAL,LOG_INFO,"Eulerian Source 'solve'. Updating Eulerian field via Solver...\n");
347  ierr = Flow_Solver(simCtx); CHKERRQ(ierr);
348  }
349  LOG_ALLOW(GLOBAL, LOG_INFO, "Eulerian Field Updated ...\n");
350  // =================================================================
351  // 3. LAGRANGIAN PARTICLE STEP
352  // =================================================================
353 
354  if (simCtx->np > 0) {
355  LOG_ALLOW(GLOBAL, LOG_INFO, "Updating Lagrangian particle system...\n");
356 
357  // a. Advect particles using the velocity interpolated from the *previous* step.
358  // P(t_{n+1}) = P(t_n) + V_p(t_n) * dt
359  ierr = UpdateAllParticlePositions(user); CHKERRQ(ierr);
360 
361  // b. Settle all particles: find their new host cells and migrate them across ranks.
362  ierr = LocateAllParticlesInGrid_TEST(user, simCtx->bboxlist); CHKERRQ(ierr);
363 
364  // c. Remove any particles that are now lost or out of the global domain.
365  ierr = CheckAndRemoveLostParticles(user, &removed_local_lost, &removed_global_lost); CHKERRQ(ierr);
366  ierr = CheckAndRemoveOutOfBoundsParticles(user, &removed_local_ob, &removed_global_ob, simCtx->bboxlist); CHKERRQ(ierr);
367  if (removed_global_lost + removed_global_ob > 0) {
368  LOG_ALLOW(GLOBAL, LOG_INFO, "Removed %d particles globally this step.\n", removed_global_lost + removed_global_ob);
369  }
370 
371  // d. Interpolate the NEW fluid velocity (just computed by Flow_Solver) onto the
372  // particles' new positions. This gives them V_p(t_{n+1}) for the *next* advection step.
373  ierr = InterpolateAllFieldsToSwarm(user); CHKERRQ(ierr);
374 
375  // e. (For Two-Way Coupling) Scatter particle data back to the grid to act as a source term.
376  ierr = CalculateParticleCountPerCell(user); CHKERRQ(ierr);
377  ierr = ScatterAllParticleFieldsToEulerFields(user); CHKERRQ(ierr);
378  }
379 
380  // =================================================================
381  // 4. UPDATE HISTORY & I/O
382  // =================================================================
383 
384  // Copy the newly computed fields (Ucont, P, etc.) to the history vectors
385  // (_o, _rm1) to prepare for the next time step.
386  for (PetscInt bi = 0; bi < simCtx->block_number; bi++) {
387  ierr = UpdateSolverHistoryVectors(&user[bi]); CHKERRQ(ierr);
388  }
389 
390  // Handle periodic file output
391  if (OutputFreq > 0 && (step + 1) % OutputFreq == 0) {
392  LOG_ALLOW(GLOBAL, LOG_INFO, "Writing output for step %d.\n",simCtx->step);
393  for (PetscInt bi = 0; bi < simCtx->block_number; bi++) {
394  ierr = WriteSimulationFields(&user[bi]); CHKERRQ(ierr);
395  }
396  if (simCtx->np > 0) {
397  ierr = WriteAllSwarmFields(user); CHKERRQ(ierr);
398  if(get_log_level() >= LOG_INFO){
399  LOG_PARTICLE_FIELDS(user,simCtx->LoggingFrequency);
400  }
401  }
402  }
403 
404  ProfilingLogTimestepSummary(simCtx->step);
405 
406  // Update Progress Bar
407  if(simCtx->rank == 0) {
408  PrintProgressBar(step,StartStep,StepsToRun,simCtx->ti);
409  if(get_log_level()>=LOG_WARNING) PetscPrintf(PETSC_COMM_SELF,"\n");
410  }
411  } // --- End of Time-Marching Loop ---
412 
413  // After the loop, print the 100% complete bar on rank 0 and add a newline
414  // to ensure subsequent terminal output starts on a fresh line.
415  if (simCtx->rank == 0 && StepsToRun > 0) {
416  PrintProgressBar(StartStep + StepsToRun - 1, StartStep, StepsToRun, simCtx->ti);
417  PetscPrintf(PETSC_COMM_SELF, "\n");
418  fflush(stdout);
419  }
420 
421  LOG_ALLOW(GLOBAL, LOG_INFO, "Time marching completed. Final time t=%.4f.\n", simCtx->ti);
422  PROFILE_FUNCTION_END;
423  PetscFunctionReturn(0);
424 }
