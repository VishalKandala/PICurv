879 /**
880  * @brief Computes the trilinear Interpolatetion weights from the Interpolatetion coefficients.
881  *
882  * This function computes the weights for trilinear Interpolatetion at the eight corners of a cell
883  * using the Interpolatetion coefficients provided along the x, y, and z directions.
884  *
885  * @param[in] a1 Interpolation coefficient along the x-direction (normalized coordinate within the cell).
886  * @param[in] a2 Interpolation coefficient along the y-direction (normalized coordinate within the cell).
887  * @param[in] a3 Interpolation coefficient along the z-direction (normalized coordinate within the cell).
888  * @param[out] w Array of 8 weights, each corresponding to one corner of the cell.
889  *
890  * @note
891  * - The coefficients `a1`, `a2`, and `a3` should be in the range [0, 1].
892  * - The order of weights corresponds to the eight corners of a hexahedral cell.
893  */
894 static inline void ComputeTrilinearWeights(PetscReal a1, PetscReal a2, PetscReal a3, PetscReal *w) {
895  LOG_ALLOW(GLOBAL, LOG_DEBUG, "Computing weights for a1=%f, a2=%f, a3=%f.\n", a1, a2, a3);
896 
897  // Ensure a1, a2, a3 are within [0,1]
898  a1 = PetscMax(0.0, PetscMin(1.0, a1));
899  a2 = PetscMax(0.0, PetscMin(1.0, a2));
900  a3 = PetscMax(0.0, PetscMin(1.0, a3));
901 
902  const PetscReal oa1 = 1.0 - a1;
903  const PetscReal oa2 = 1.0 - a2;
904  const PetscReal oa3 = 1.0 - a3;
905 
906  w[0] = oa1 * oa2 * oa3; /* cornerOffsets[0] => (0,0,0) */
907  w[1] = a1 * oa2 * oa3; /* cornerOffsets[1] => (1,0,0) */
908  w[2] = oa1 * a2 * oa3; /* cornerOffsets[2] => (0,1,0) */
909  w[3] = a1 * a2 * oa3; /* cornerOffsets[3] => (1,1,0) */
910  w[4] = oa1 * oa2 * a3; /* cornerOffsets[4] => (0,0,1) */
