# --- Docs targets ---
DOCS_DIR := docs
DOCS_OUT := $(DOCS_DIR)/docs_build/html

.PHONY: docs open-docs clean-docs

docs:
	@echo "==> Generating Doxygen docs"
	@cd $(DOCS_DIR) && doxygen Doxyfile
	@echo "HTML: $(DOCS_OUT)/index.html"

open-docs: docs
ifeq ($(shell uname),Darwin)
	open $(DOCS_OUT)/index.html
else
	xdg-open $(DOCS_OUT)/index.html || true
endif

clean-docs:
	@rm -rf $(DOCS_DIR)/docs_build $(DOCS_DIR)/doxygen.warnings
